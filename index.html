async function syncWithGitHub(isSilent = false) { const config = getCloudConfig(); if (!config.user || !config.repo || !config.token) return; const url = `https://api.github.com/repos/${config.user}/${config.repo}/contents/ledger_db.json`; const headers = { "Authorization": `token ${config.token}`, "Accept": "application/vnd.github.v3+json", "Content-Type": "application/json" }; if (!isSilent) document.getElementById('syncBtn').innerText = "⚡ 正在对账..."; try { // 1. 获取最新状态 (获取 SHA) let response = await fetch(url, { headers }); let cloudSha = null; let cloudData = null; if (response.ok) { let json = await response.json(); cloudSha = json.sha; let decodedText = decodeURIComponent(escape(atob(json.content))); cloudData = JSON.parse(decodedText); } // 2. 合并数据 (解决闪回的关键：保留逻辑) if (cloudData) { // 合并账户余额 (若有必要) let accountMap = new Map(storage.accounts.map(a => [a.name, a])); (cloudData.accounts || []).forEach(a => accountMap.set(a.name, a)); storage.accounts = Array.from(accountMap.values()); // 合并账单 (以本地删除状态为准) let txMap = new Map(storage.transactions.map(t => [t.id, t])); (cloudData.transactions || []).forEach(t => { if (!txMap.has(t.id) || (t.updatedAt > (txMap.get(t.id)?.updatedAt || 0))) { txMap.set(t.id, t); } }); storage.transactions = Array.from(txMap.values()).filter(t => !t.isDeleted); storage.transactions.sort((a, b) => new Date(b.date) - new Date(a.date)); } // 3. 上传数据 storage.sha = cloudSha; let contentToUpload = btoa(unescape(encodeURIComponent(JSON.stringify(storage, null, 2)))); let putBody = { message: "☁️ 桃气紫账本数据同步", content: contentToUpload }; if (storage.sha) putBody.sha = storage.sha; let putResponse = await fetch(url, { method: "PUT", headers: headers, body: JSON.stringify(putBody) }); if (putResponse.ok) { let putJson = await putResponse.json(); storage.sha = putJson.content.sha; saveToStorage(); if (!isSilent) alert("✨ 恭喜！数据同步成功！"); } else { console.error("同步失败，请检查 Token 权限或网络"); } } catch (error) { console.error("同步异常:", error); } finally { document.getElementById('syncBtn').innerText = "☁️ 手动同步"; refreshAllUI(); } }
